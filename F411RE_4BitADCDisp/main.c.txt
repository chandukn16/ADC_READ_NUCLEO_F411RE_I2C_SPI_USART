/**
  * @brief  The application entry point.
  * @retval int
  */
int main(void)
{

  /* USER CODE BEGIN 1 */

  /* USER CODE END 1 */

  /* MCU Configuration--------------------------------------------------------*/

  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
  HAL_Init();

  /* USER CODE BEGIN Init */

  /* USER CODE END Init */

  /* Configure the system clock */
  SystemClock_Config();

  /* USER CODE BEGIN SysInit */

  /* USER CODE END SysInit */

  /* Initialize all configured peripherals */
  MX_GPIO_Init();
  MX_ADC1_Init();
  MX_SPI3_Init();
  /* USER CODE BEGIN 2 */
  // LED blink during startup
  for (int i=0; i<10; i++) {
      HAL_GPIO_TogglePin(LED_GPIO_Port, LED_Pin);
      HAL_Delay(100);
  }

  /* Initialize display driver */
  DISP_Init(&hspi3, RCLK_GPIO_Port, RCLK_Pin);
  DISP_SetBrightness(80); /* 0..100 (higher = brighter) */

  // Display startup sequence
  DISP_SetDigits(8, 8, 8, 8);
  HAL_Delay(1000);
  DISP_SetDigits(1, 2, 3, 4);
  HAL_Delay(1000);


  /* USER CODE END 2 */

  /* Infinite loop */
  /* USER CODE BEGIN WHILE */
  while (1)
  {
    /* USER CODE END WHILE */

    /* USER CODE BEGIN 3 */
	  uint32_t current_time = HAL_GetTick();

	     /* Refresh display continuously - non-blocking */
	     DISP_RefreshOnce();

	     /* Read ADC at regular intervals without blocking display */
	     if (current_time - last_adc_read_time >= ADC_READ_INTERVAL) {
	         /* Start ADC conversion */
	         HAL_ADC_Start(&hadc1);

	         if (HAL_ADC_PollForConversion(&hadc1, 5) == HAL_OK)
	         {
	             adc_raw = (uint16_t)HAL_ADC_GetValue(&hadc1); // 0..4095 for 12-bit ADC

	             /* Cap to 4 digits (9999) just in case */
	             if (adc_raw > 9999) adc_raw = 9999;

	             /* split into digits: thousand, hundred, ten, unit */
	             uint8_t d3 = (adc_raw / 1000) % 10;
	             uint8_t d2 = (adc_raw / 100)  % 10;
	             uint8_t d1 = (adc_raw / 10)   % 10;
	             uint8_t d0 = (adc_raw / 1)    % 10;

	             /* Note ordering: DISP_SetDigits(d0,d1,d2,d3) maps to digit0..digit3 in code */
	             DISP_SetDigits(d0, d1, d2, d3);
	         }
	         last_adc_read_time = current_time;

	         /* Toggle LED to indicate ADC reading */
	         HAL_GPIO_TogglePin(LED_GPIO_Port, LED_Pin);
	     }

  }
  /* USER CODE END 3 */
}
